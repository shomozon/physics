\begin{eqnarray} m\boldsymbol{a} = \boldsymbol{f} \Leftrightarrow m\boldsymbol{a}\cdot \boldsymbol{v} = \boldsymbol{f}\cdot \boldsymbol{v} \\ (左辺) = m\dfrac{d\boldsymbol{v}}{dt}\cdot \boldsymbol{v} \\ = m(\dfrac{d v_{x}}{dt}v_{x} + \dfrac{d v_{y}}{dt}v_{y} + \dfrac{d v_{z}}{dt}v_{z}) = m\dfrac{d}{dt}(\dfrac{v_{x}^{2}}{2} + \dfrac{v_{y}^{2}}{2} + \dfrac{v_{z}^{2}}{2}) = m\dfrac{d}{dt}(\dfrac{v^2}{2}) = \dfrac{d}{dt}(\dfrac{1}{2} m v^2) \end{eqnarray}
